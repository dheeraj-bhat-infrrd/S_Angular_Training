Copy jar to the server
	pscp -i E:\raremile_projects\socialsurvey\info-pem.ppk E:\raremile_projects\socialsurvey\repo\Source\SSCore\target\SSCore-1.0.jar ec2-user@52.11.208.165:/home/ec2-user/deliverables/SSCore-1.0.jar

Login to prod using putty with user: ec2-user

take the backup of the jar and then move the new jar 
	sudo cp deliverables/SSCore-1.0.jar deliverables/backup/SSCore-1.0.jar_<todays date>
	sudo mv deliverables/SSCore-1.0.jar /var/lib/batch/lib/SSCore-1.0.jar

delete batch job process
	ps -ef | grep java

go to the bin folder
	cd /var/lib/batch/bin/

delete lock file
	sudo rm -f batchstarter.lock 

start batch process
	sudo ./batchstarter.sh &
